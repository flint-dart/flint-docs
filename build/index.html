import 'package:flint_dart/flint_dart.dart';

void main() async {
  final app = Flint();
  
  // Database configuration
  app.db.addConnection(DatabaseConfig(
    driver: 'mysql',
    host: 'localhost',
    database: 'my_app',
    username: 'user',
    password: 'password',
  ));
  
  // Authentication
  app.auth.addProvider(GoogleAuthProvider());
  
  // Routes
  app.get('/', (req, res) {
    return res.json({'message': 'Hello FlintDart!'});
  });
  
  // Start server
  await app.serve(port: 3000);
}

Installation and basic setup is quick and easy

Install FlintDart

dart pub add flint_dart
2

Create Your App

dart flint_dart create my_app

3

Run Development Server

dart run bin/hot_reload.dart
4

Visit Your App

Open your browser and navigate to http://localhost:3000

Routing

// Define routes with ease
app.get('/users/{id}', (req, res) async {
  final user = await User.find(req.params['id']);
  return res.json(user);
});

app.post('/users', (req, res) async {
  final user = await User.create(req.body);
  return res.json(user, status: 201);
});

Database Models

class User extends Model {
  String name;
  String email;
  
  // Relationships
  BelongsTo<Company> company;
  HasMany<Post> posts;
  
  // Scopes
  static Query<User> active() {
    return User.where('status', 'active');
  }
}

Authentication

// Protect routes
app.get('/dashboard', auth(), (req, res) {
  return res.view('dashboard', {
    'user': req.auth.user
  });
});

// Social login
app.get('/auth/google', 
  auth().redirect('google'));
